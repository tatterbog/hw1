#include <iostream>
#include <vector>

template<typename T>
int linearSearch(const std::vector<T>& vec, const T& el) {
    int len = vec.size();
    for(int i = 0; i < len; i++){
        if(vec[i] == el){
            return i;
        }
    }
    return -1;
}

int main() {
    
    std::vector<int> arr1 = {1, 2, 3, 4};
    std::vector<double> arr2 = {1.2, 2.1, 3.4, 4.4};
    std::vector<std::string> arr3 = {"He", "llo", " wor", "ld"};
    
    std::cout << linearSearch(arr1, 1) << "\n";
    std::cout << linearSearch(arr2, 5.3) << "\n";
    std::cout << linearSearch(arr3, std::string("ld")) << "\n";
    
    return 0;
}
