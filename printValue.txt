#include <iostream>

template<typename T>
void printElement(const T elem) {
    std::cout << elem << '\n';
}

template<>
void printElement(bool elem) {
    std::cout << std::boolalpha << elem << '\n';
}

template<>
void printElement(const char* elem) {
    std::cout << '"' << elem << '"' << '\n';
}

int main() {
    printElement<bool>(0);
    printElement<double>(10.2);
    printElement<const char *>("CPP");
    return 0;
}
